(build containers)
sudo docker-compose -f docker-compose-dev.yml build

(up to run container, -d to run in background)
sudo docker-compose -f docker-compose-dev.yml up -d

(update container)
sudo docker-compose -f docker-compose-dev.yml up -d --build

(acces database through psql)
sudo docker-compose -f docker-compose-dev.yml exec users-db psql -U postgres

(run manage.py command)
sudo docker-compose -f docker-compose-dev.yml run users python manage.py recreate-db
